Table Section
CREATE TABLE
- create a table
  Create table "NAME" (
    "NAME" "DATA_TYPE" "CONSTRAINTS"
  )

- Example Line
  Create table Products (                          {======Comments======} 
  ID int AUTO_INCREMENT PRIMARY KEY,               # Automatic NUmbering Column
  ProductName varchar(100) NOT NULL UNIQUE,        # Needs a name and Must be different from others
  Description text,                                # No specific text-length limit
  Quantity int CHECK (Quantity >= 0),              # Checks the data that it must not be negative
  Price decimal(10, 2) NOT NULL CHECK (Price > 0), # Example - PhP 99.99, no negatives
  IsAvailable boolean DEFAULT TRUE,                # TRUE OR FALSE, Default value is true
  LastUpdated timestamp DEFAULT CURRENT_TIMESTAMP  # Sets what time was updated
  );

- Constraints in Columns
  PRIMARY KEY    - Both NOT NULL and UNIQUE, A table can only have 1 PRIMARY KEY
  FOREIGN KEY    - Adds link between 2 tables using the PRIMARY KEY
  UNIQUE         - All data must be different from one another
  NOT NULL       - All data in this column must not be null and should have value
  DEFAULT        - If theres no value added to it, it automatically sets that to specifed default value
  CHECK          - Checks if the data is correct depends on the condition
  AUTO_INCREMENT - Adds numbering 1..2..3..4..etc

- Data types
  INT                    - integer
  DECIMAL(digits, decim) - integer but with decimal (digits, digits after decimal)
  DOUBLE/FLOAT           - numerical integer
  VARCHAR(size)          - string with adjustable size of max number of characters
  CHAR(size)             - string with fixed size of number of characters
  TEXT                   - string thats used for larger text
  DATE                   - date types - "YYYY-MM-DD" (2005-08-25)
  TIME                   - clock time - "HH:MM:SS" (12:50:00)
  DATETIME               - both types - "YYYY-MM-DD HH:MM:SS"
  TIMESTAMP              - timestamp  - same format of datetime
  BOOL / BOOLEAN         - TRUE OR FALSE
  
ALTER TABLE
- Add a column
  Alter Table "TABLE_NAME"
  ADD "COLUMN_NAME" "DATA_TYPE" "CONSTRAINTS";

- Delete a column
  Alter Table "TABLE_NAME"
  DROP COLUMN "COLUMN_NAME"; 

- Change/Modify a column
  Alter table "TABLE_NAME"
  MODIFY COLUMN "COLUMN_NAME" "DATA_TYPE" "CONSTRAINTS";

- Change name of column
  Alter table "TABLE_NAME"
  CHANGE "OLD_NAME" "NEW_NAME" "DATA_TYPE";

- Change name of table 
  Alter table "TABLE_NAME"
  RENAME to "NEW_NAME";

DROP TABLE
- Delete a table
  DROP TABLE "TABLE_NAME";

TRUNCATE TABLE
- Delete all data inside the table but not the structure
  TRUNCATE TABLE "TABLE_NAME";

UPDATE 
- update the date on the table
  UPDATE "TABLE NAME"
  SET "COLUMN_NAME" = "NEW_DATA_VALUE", "COLUMN_NAME" = "NEW_DATA_VALUE"
  WHERE "COLUMN_NAME" = "DATA_VALUE";

DELETE FROM
- delete a data
  DELETE FROM "TABLE_NAME"
  WHERE "COLUMN_NAME" = "DATA_VALUE";

SELECT
- get the data from table                {======Comments======}
  SELECT "COLUMN_NAME" or *(all)         # select a column or * all what to show 
  FROM "TABLE_NAME"                      # which table is it?
  WHERE "CONDITION"                      # what condition it needs to show that column
  GROUP BY "COLUMN NAME"                 # group the results 
  HAVING "CONDITION"                     # condition to filter the groups 
  ORDER BY "COLUMN_NAME" ASC/DESC        # ascending or decending order
  LIMIT "NUMBER";                        # limits the number or rows

Example Line
  SELECT ProductName, Quantity, Price
  FROM Products
  WHERE IsAvailable = TRUE
  ORDER BY Price DESC;

Conditions

  <=       - Less/Equal
 <> or !=  - Not Equal

Logical Operators
